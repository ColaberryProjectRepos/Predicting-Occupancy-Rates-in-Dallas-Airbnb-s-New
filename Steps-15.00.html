Creating DAX Measures 15

Creating the necessary measures

1.     Click on FinalFabricFile table and select New Measure

2.     Start writing the DAX formula (all the formulas are at the bottom) or copy and paste one by one.

3.     After you done click on the green right mark so it will be saved.

4.     Repeat the steps until all the measures are created.

 

.TotalProperty = COUNTROWS(VALUES(FinalFabricFile[id_x]))

 

.AvgPricePerNight = AVERAGE('FinalFabricFile'[price_x])

 

.OccupRate = AVERAGEX(FinalFabricFile,(365 - FinalFabricFile[availability_365_x]) / 365)

 

.Property YOY =

VAR CurrentYear = MAX('CalendarTable'[Date].[Year])

VAR CurrentYearProperty = [.TotalProperty]

VAR PrvYear = CurrentYear - 1

VAR PrvYearproperty =

    CALCULATE(

        [.TotalProperty],

        FILTER(

            ALL('CalendarTable'[Date].[Year]),  -- Remove any date filters for accurate comparison

            'CalendarTable'[Date].[Year] = PrvYear

        )

    )

RETURN

IF(

    ISBLANK(PrvYearproperty),

    BLANK(),  -- Return a blank if no data for the previous year

    DIVIDE(CurrentYearProperty - PrvYearproperty, PrvYearproperty)

)

 

 

.Avg price YOY =

VAR CurrentYear = MAX('CalendarTable'[Date].[Year])

VAR CurrentYearAvgPrice = [.AvgPricePerNight]

VAR PrvYear = CurrentYear - 1

VAR PrvYearAvgPrice =

    CALCULATE(

        [.AvgPricePerNight],

        FILTER(

            ALL('CalendarTable'[Date].[Year]),  -- Remove any date filters for accurate comparison

            'CalendarTable'[Date].[Year] = PrvYear

        )

    )

RETURN

IF(

    ISBLANK(PrvYearAvgPrice),

    BLANK(),  -- Return a blank if no data for the previous year

    DIVIDE(CurrentYearAvgPrice - PrvYearAvgPrice, PrvYearAvgPrice)

)
